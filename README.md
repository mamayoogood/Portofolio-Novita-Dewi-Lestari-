        // Fungsi untuk upload sertifikat
        document.getElementById('uploadCert1').addEventListener('change', function(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById('cert1').src = e.target.result;
                };
                reader.readAsDataURL(file);
            }
        });

        document.getElementById('uploadCert2').addEventListener('change', function(event) {

            if (file) {

                    document.getElementById('cert2').src = e.target.result;
                };
                reader.readAsDataURL(file);
            }
        });

        // Fungsi untuk upload foto galeri
        function uploadPhotos() {
            const files = document.getElementById('uploadPhoto').files;
            const gallery = document.getElementById('photoGallery');
            for (let i = 0; i < files.length; i++) {
                const file = files[i];
                const reader = new FileReader();
                reader.onload = function(e) {
                    const img = document.createElement('img');
                    img.src = e.target.result;
                    gallery.appendChild(img);
                };
                reader.readAsDataURL(file);
            }
        }
    </script>
